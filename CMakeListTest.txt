CC = gcc
OBJS_EVENT_MANAGER = date.o event_manager.o event_manager_tests.o libpriority_queue.a
OBJS_PRIORITY_QUEUE = priority_queue.o priority_queue_tests.o
COMP_FLAG = -std=c99 -Wall -Werror --pedantic-errors -DNDEBUG
LIB_FLAGS = -L. -lpriority_queue
EXEC_EVENT = event_manager
EXEC_PRIORITY_QUEUE = priority_queue

$(EXEC_EVENT): $(OBJS_EVENT_MANAGER)
	$(CC) $(COMP_FLAG) $(OBJS_EVENT_MANAGER) $(LIB_FLAGS) -o event_manager

$(EXEC_PRIORITY_QUEUE): $(OBJS_PRIORITY_QUEUE)
	$(CC) $(COMP_FLAG) $(OBJS_PRIORITY_QUEUE) -o priority_queue

date.o: date.c date.h
	$(CC) -c $(COMP_FLAG) $*.c

event_manager.o: event_manager.c event_manager.h date.h priority_queue.h
	$(CC) -c $(COMP_FLAG) $*.c

event_manager_tests.o: tests/event_manager_tests.c tests/test_utilities.h event_manager.h date.h
	$(CC) -c $(COMP_FLAG) tests/event_manager_tests.c

priority_queue.o: priority_queue.c priority_queue.h
	$(CC) -c $(COMP_FLAG) $*.c

priority_queue_tests.o: tests/priority_queue_tests.c tests/test_utilities.h priority_queue.h
	$(CC) -c $(COMP_FLAG) tests/priority_queue_tests.c

clean:
	rm -f $(OBJS_EVENT_MANAGER) $(OBJS_PRIORITY_QUEUE) $(EXEC_PRIORITY_QUEUE) $(EXEC_EVENT)
